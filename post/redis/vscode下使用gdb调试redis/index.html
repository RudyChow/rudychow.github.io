Vscode下使用GDB调试Redis

December 13, 2020 · Rudy
  1. 配置.vscode/launch.json：
{
    "version": "0.2.0",
    "configurations": [
        {
            "name": "redis启动",
            "type": "cppdbg",
            "request": "launch",
            "program": "${workspaceFolder}/src/redis-server",
            "args": ["${workspaceFolder}/redis.conf"],
            "stopAtEntry": false,
            "cwd": "${workspaceFolder}",
            "environment": [],
            "externalConsole": false,
            "MIMode": "gdb",
            "preLaunchTask": "make redis",
            "setupCommands": [
                {
                    "description": "为 gdb 启用整齐打印",
                    "text": "-enable-pretty-printing",
                    "ignoreFailures": true
                }
            ]
        }
    ]
}
  1. 配置.vscode/tasks.json：
{
    "version": "2.0.0",
    "tasks": [
        {
            "label": "make redis",
            "type": "shell",
            "command": "make",
            "args": [
                "CFLAGS=\"-g -O0\""
            ],
            "options": {
                "cwd": "${workspaceFolder}/src"
            }
        }
    ]
}
  1. Start Debugging
